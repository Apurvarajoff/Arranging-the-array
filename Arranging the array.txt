void Rearrange(int arr[], int n)
{
    // Your code goes here
    
   int temp[n];
    int neg = 0;
    
    for (int i = 0; i < n; i++) {
        if (arr[i] < 0)
            neg++;
    }
    
    int l = 0, r = neg;
    for (int i = 0; i < n; i++) {
        if (arr[i] < 0) {
            temp[l] = arr[i];
            l++;
        } else {
            temp[r] = arr[i];
            r++;
        }
    }
    
    for (int i = 0; i < n; i++) {
        arr[i] = temp[i];
    }
}
